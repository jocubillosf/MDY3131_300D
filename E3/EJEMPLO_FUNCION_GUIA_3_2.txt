CREATE OR REPLACE FUNCTION FN_RESPONSABLE_PGC(P_ID_EDIFICIO NUMBER, P_NRO_DEPTO NUMBER) RETURN VARCHAR2 AS
    V_RESPONSABLE_PGC VARCHAR2(50);
BEGIN
    SELECT DISTINCT(RPGC.PNOMBRE_RPGC || ' ' || RPGC.SNOMBRE_RPGC || ' ' 
    || RPGC.APPATERNO_RPGC || ' ' || RPGC.APMATERNO_RPGC) NOMBRE_RESPONSABLE_PAGO
    INTO V_RESPONSABLE_PGC
    FROM GASTO_COMUN GC
    JOIN RESPONSABLE_PAGO_GASTO_COMUN RPGC
    ON GC.NUMRUN_RPGC = RPGC.NUMRUN_RPGC
    WHERE GC.ID_EDIF = P_ID_EDIFICIO AND GC.NRO_DEPTO = P_NRO_DEPTO;
    RETURN V_RESPONSABLE_PGC;
END FN_RESPONSABLE_PGC;

DECLARE
BEGIN
    DBMS_OUTPUT.PUT_LINE(FN_RESPONSABLE_PGC(20,408));
END;
